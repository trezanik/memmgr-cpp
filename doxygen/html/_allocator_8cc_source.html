319 Allocator::TrackedAlloc(
320  const uint32_t num_bytes,
321  const char* file,
322  const char* function,
323  const uint32_t line,
324  const Object* owner
325 )
326 {
327  memblock_header* mem_block = nullptr;
328  memblock_footer* mem_footer = nullptr;
329  void* mem_return = nullptr;
330  uint32_t patched_alloc = 0; // num_bytes + memblocks
331 
332  // allocate the requested amount, plus the size of the header & footer memblocks
333  patched_alloc = num_bytes + HEADER_FOOTER_SIZE;
334 
335  // the actual, real, physical allocation of memory
336  mem_block = (memblock_header*)malloc(patched_alloc);
337 
338  if ( mem_block == nullptr )
339  goto alloc_failure;
340 
341  // initialize the value for the new memory
342  memset(mem_block, MEM_ON_INIT, patched_alloc);
343 
344 #if defined(_WIN32)
345 # define PATH_CHAR '\\'
346 #else
347 # define PATH_CHAR '/'
348 #endif
349  // we don't want the full path information that compilers set
350  if (( file = strrchr(file, PATH_CHAR)) != nullptr )
351  file++;
352 
353  // calculate the offsets of the return memory and the footer
354  mem_return = block_offset_realmem(mem_block);
355  mem_footer = block_offset_footer(mem_block, num_bytes);
356 
357  // prepare the structure internals
358  mem_footer->magic = mem_footer_magic;
359  mem_block->footer = mem_footer;
360  mem_block->magic = mem_header_magic;
361  mem_block->owner = owner;
362  mem_block->line = line;
363  mem_block->real_size = patched_alloc;
364  mem_block->requested_size = num_bytes;
365 
366  /* you should be using strlcpy if you're doing character buffer copying
367  * and other functionality!! For the sake of securing your application,
368  * please USE IT!
369  * We provide a 'normal, well-known' alternative for the sake of the
370  * example using strncpy (which is in standard headers everywhere) */
371 #if defined(HAVE_STRLCPY)
372  strlcpy(mem_block->file, file, sizeof(mem_block->file));
373  strlcpy(mem_block->function, function, sizeof(mem_block->function));
374 #else
375  strncpy(mem_block->file, file, sizeof(mem_block->file)-1);
376  strncpy(mem_block->function, function, sizeof(mem_block->function)-1);
377 #endif
378 
379  /* lock this class, only 1 thread to update sensitive internals at a
380  * time - lock for as little time as possible! */
381  _mutex.lock();
382 
383  // update the stats, using patched values
384  _allocs++;
385  _current_allocated += patched_alloc;
386  _total_allocated += patched_alloc;
387  // append it to the vector
388  _memblocks.push_back(mem_block);
389 
390  // unlock the allocator, other threads can now allocate from this class
391  _mutex.unlock();

402 Allocator::TrackedFree(
403  void* memory
404 )
405 {
406  memblock_header* mem_block = nullptr;
407 
408  // as per the C standard, if it's a nullptr, do nothing
409  if ( memory == nullptr )
410  return;
411 
412  if ( !ValidateMemory(memory) )
413  return;
414 
415  mem_block = block_offset_header(memory);
416 
417 #if !defined(DISABLE_MEMORY_OP_TO_STDOUT)
418  printf( "free [%s (%u bytes) line %u]\n"
419  "\tBlock: %p | Usable Block: %p\n",
420  mem_block->file, mem_block->requested_size, mem_block->line,
421  mem_block, memory);
422 #endif
423 
424  // stop other class modifications
425  _mutex.lock();
426 
427  // remove the mem_block from the vector
428  _memblocks.erase(std::find(_memblocks.begin(), _memblocks.end(), mem_block));
429  // update the context stats
430  _frees++;
431  _current_allocated -= (mem_block->real_size);
432 
433  // we're done with the class internals, open it up again
434  _mutex.unlock();
435 
436  // fill the app-allocated memory (highlights use after free)
437  memset(mem_block, MEM_AFTER_FREE, mem_block->real_size);
438  // perform the actual freeing of memory, including our header + footer
439  free(mem_block);
440 
441  return;
